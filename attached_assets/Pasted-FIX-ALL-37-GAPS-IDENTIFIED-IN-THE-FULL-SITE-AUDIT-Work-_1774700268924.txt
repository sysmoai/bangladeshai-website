const BASE_URL = 'https://bangladeshai.org'

  'workforce-education', 'risk-management'
]

  'bangladesh-ai-readiness-47', 'why-ai-is-not-magic', 'ai-bangla-language-guide',
  'economic-impact-analysis', 'what-singapore-teaches-us', 'bangla-nlp-sovereign-stack',
  'youth-and-ai-jobs', 'population-asset-not-problem', 'sovereign-ai-infrastructure-why-it-matters'
]

export default function sitemap(): MetadataRoute.Sitemap {
  const locales = ['en', 'bn']
  const entries: MetadataRoute.Sitemap = []

  // Static routes
  for (const route of staticRoutes) {
    for (const locale of locales) {
      entries.push({
        url: `${BASE_URL}/${locale}${route === '/' ? '' : route}`,
        lastModified: new Date('2026-03-27'),
        changeFrequency: route === '/' ? 'weekly' : 'monthly',
        priority: route === '/' ? 1.0 : 0.7,
      })
    }
  }

  // Research pages
  for (const slug of researchSlugs) {
    for (const locale of locales) {
      entries.push({
        url: `${BASE_URL}/${locale}/research/${slug}`,
        lastModified: new Date('2026-03-27'),
        changeFrequency: 'monthly',
        priority: 0.9,
      })
    }
  }

  // Blog posts
  for (const slug of blogSlugs) {
    for (const locale of locales) {
      entries.push({
        url: `${BASE_URL}/${locale}/blog/${slug}`,
        lastModified: new Date('2026-03-27'),
        changeFrequency: 'monthly',
        priority: 0.8,
      })
    }
  }

  return entries
}
